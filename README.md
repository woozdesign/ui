## üî® Usage

```jsx
import React from 'react';
import { Button, Layout } from 'woozdesign-react';

const App = () => (
  <Layout.Container>
    <Layout.Row>
      <Layout.Col xs={24}>
        <Button type="primary">PRESS ME</Button>
      </Layout.Col>
    </Layout.Row>
  </Layout.Container>
);
```
